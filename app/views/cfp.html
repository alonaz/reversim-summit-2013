require 'speaker'
require 'israel'

module Israel
  module Summit
    def call_for_papers(speaker, subject)
      apply(speaker) if ["DevOps, NoOps, CD, Dev Culture",
                         "Software Craftsmanship",
                         "Mobile dev",
                         "Web dev",
                         "Product, Lean Startup, UX"
                        ].include(subject)
                        and (speaker.available?(Date.new(2013, 2, 18))
                           or speaker.available?(Date.new(2013, 2, 19)))
    end
    def apply(speaker)
      # We prefer github, but also settle for goo-ol emails
      if speaker.can_github?
        speaker.make_pull_request('/rantav/reversim-summit-2013')
      else
        spekaer.send_mail('rantav@ the gmail')
    end
  end
end
